// U88_BasicProtection.pas - 纯代码实现，无第三方组件依赖
unit U88_BasicProtection;

interface

uses
  Windows, SysUtils, Classes, IdHMACSHA256, IdGlobal;

type
  TBasicProtection = class
  private
    class function GetDynamicKey: string;
    class function GenerateRandomIV: TBytes;
    class function BytesToHex(const ABytes: TBytes): string;
    class function HexToBytes(const AHex: string): TBytes;
  public
    class function EncryptSensitiveData(const AData: string): string;
    class function DecryptSensitiveData(const AEncryptedData: string): string;
    class function CalculateHMAC(const AData: string): string;
    class function VerifyDataIntegrity(const AData, AHMAC: string): Boolean;
    class function CalculateFileHash(const AFileName: string): string;
  end;

implementation

// 动态密钥生成（避免硬编码）
class function TBasicProtection.GetDynamicKey: string;
var
  Part1, Part2, Part3: string;
  ExePath: string;
begin
  ExePath := Application.ExeName;
  Part1 := ReverseString(ExtractFileName(ExePath));
  Part2 := IntToStr(FileAge(ExePath)); // 基于文件时间戳
  Part3 := 'SecureKey2024';
  Result := Copy(Part1 + Part2 + Part3, 1, 32); // 确保32字节密钥
  
  // 填充到32字节
  while Length(Result) < 32 do
    Result := Result + '0';
end;

// 生成随机IV (使用Windows CryptoAPI)
class function TBasicProtection.GenerateRandomIV: TBytes;
var
  hProv: HCRYPTPROV;
begin
  SetLength(Result, 16); // AES块大小为16字节
  
  if CryptAcquireContext(hProv, nil, nil, PROV_RSA_FULL, CRYPT_VERIFYCONTEXT) then
  try
    if not CryptGenRandom(hProv, 16, @Result[0]) then
      raise Exception.Create('生成随机IV失败');
  finally
    CryptReleaseContext(hProv, 0);
  end
  else
    raise Exception.Create('获取加密上下文失败');
end;

// AES-256-CBC加密 (使用Windows CryptoAPI)
class function TBasicProtection.EncryptSensitiveData(const AData: string): string;
var
  hProv: HCRYPTPROV;
  hKey: HCRYPTKEY;
  hHash: HCRYPTHASH;
  DataBytes, EncryptedData, IV: TBytes;
  DataLen: DWORD;
  Key: string;
  KeyBytes: TBytes;
begin
  Key := GetDynamicKey;
  KeyBytes := TEncoding.UTF8.GetBytes(Key);
  DataBytes := TEncoding.UTF8.GetBytes(AData);
  IV := GenerateRandomIV;
  
  // 获取AES加密上下文
  if not CryptAcquireContext(hProv, nil, MS_ENH_RSA_AES_PROV, PROV_RSA_AES, CRYPT_VERIFYCONTEXT) then
    raise Exception.Create('获取AES加密上下文失败');
  
  try
    // 创建哈希对象用于密钥派生
    if not CryptCreateHash(hProv, CALG_SHA_256, 0, 0, hHash) then
      raise Exception.Create('创建哈希对象失败');
    
    try
      // 添加密钥数据到哈希
      if not CryptHashData(hHash, @KeyBytes[0], Length(KeyBytes), 0) then
        raise Exception.Create('哈希密钥数据失败');
      
      // 从哈希派生AES-256密钥
      if not CryptDeriveKey(hProv, CALG_AES_256, hHash, CRYPT_EXPORTABLE, hKey) then
        raise Exception.Create('派生AES密钥失败');
      
      try
        // 设置CBC模式
        if not CryptSetKeyParam(hKey, KP_MODE, @CRYPT_MODE_CBC, 0) then
          raise Exception.Create('设置CBC模式失败');
        
        // 设置IV
        if not CryptSetKeyParam(hKey, KP_IV, @IV[0], 0) then
          raise Exception.Create('设置IV失败');
        
        // 准备加密数据 (预留PKCS7填充空间)
        SetLength(EncryptedData, Length(DataBytes) + 16);
        Move(DataBytes[0], EncryptedData[0], Length(DataBytes));
        DataLen := Length(DataBytes);
        
        // 执行AES-256-CBC加密
        if not CryptEncrypt(hKey, 0, True, 0, @EncryptedData[0], DataLen, Length(EncryptedData)) then
          raise Exception.Create('AES加密失败');
        
        // 调整加密数据长度
        SetLength(EncryptedData, DataLen);
        
        // 返回 IV + 加密数据 的十六进制表示
        Result := BytesToHex(IV) + '|' + BytesToHex(EncryptedData);
        
      finally
        CryptDestroyKey(hKey);
      end;
    finally
      CryptDestroyHash(hHash);
    end;
  finally
    CryptReleaseContext(hProv, 0);
  end;
end;

// AES-256-CBC解密
class function TBasicProtection.DecryptSensitiveData(const AEncryptedData: string): string;
var
  hProv: HCRYPTPROV;
  hKey: HCRYPTKEY;
  hHash: HCRYPTHASH;
  Parts: TArray<string>;
  IV, EncryptedBytes, DecryptedData: TBytes;
  DataLen: DWORD;
  Key: string;
  KeyBytes: TBytes;
begin
  Result := '';
  
  // 分离IV和加密数据
  Parts := AEncryptedData.Split(['|']);
  if Length(Parts) <> 2 then
    raise Exception.Create('加密数据格式错误');
  
  IV := HexToBytes(Parts[0]);
  EncryptedBytes := HexToBytes(Parts[1]);
  
  Key := GetDynamicKey;
  KeyBytes := TEncoding.UTF8.GetBytes(Key);
  
  // 获取AES解密上下文
  if not CryptAcquireContext(hProv, nil, MS_ENH_RSA_AES_PROV, PROV_RSA_AES, CRYPT_VERIFYCONTEXT) then
    raise Exception.Create('获取AES解密上下文失败');
  
  try
    // 创建哈希对象用于密钥派生
    if not CryptCreateHash(hProv, CALG_SHA_256, 0, 0, hHash) then
      raise Exception.Create('创建哈希对象失败');
    
    try
      // 添加密钥数据到哈希
      if not CryptHashData(hHash, @KeyBytes[0], Length(KeyBytes), 0) then
        raise Exception.Create('哈希密钥数据失败');
      
      // 从哈希派生AES-256密钥
      if not CryptDeriveKey(hProv, CALG_AES_256, hHash, CRYPT_EXPORTABLE, hKey) then
        raise Exception.Create('派生AES密钥失败');
      
      try
        // 设置CBC模式
        if not CryptSetKeyParam(hKey, KP_MODE, @CRYPT_MODE_CBC, 0) then
          raise Exception.Create('设置CBC模式失败');
        
        // 设置IV
        if not CryptSetKeyParam(hKey, KP_IV, @IV[0], 0) then
          raise Exception.Create('设置IV失败');
        
        // 准备解密数据
        DecryptedData := Copy(EncryptedBytes);
        DataLen := Length(DecryptedData);
        
        // 执行AES-256-CBC解密
        if not CryptDecrypt(hKey, 0, True, 0, @DecryptedData[0], DataLen) then
          raise Exception.Create('AES解密失败');
        
        // 调整解密数据长度并转换为字符串
        SetLength(DecryptedData, DataLen);
        Result := TEncoding.UTF8.GetString(DecryptedData);
        
      finally
        CryptDestroyKey(hKey);
      end;
    finally
      CryptDestroyHash(hHash);
    end;
  finally
    CryptReleaseContext(hProv, 0);
  end;
end;

// 字节数组转十六进制字符串
class function TBasicProtection.BytesToHex(const ABytes: TBytes): string;
var
  I: Integer;
begin
  Result := '';
  for I := 0 to Length(ABytes) - 1 do
    Result := Result + IntToHex(ABytes[I], 2);
end;

// 十六进制字符串转字节数组
class function TBasicProtection.HexToBytes(const AHex: string): TBytes;
var
  I: Integer;
begin
  SetLength(Result, Length(AHex) div 2);
  for I := 0 to Length(Result) - 1 do
    Result[I] := StrToInt('$' + Copy(AHex, I * 2 + 1, 2));
end;

// HMAC-SHA256校验 (使用Delphi内置IdHMACSHA256)
class function TBasicProtection.CalculateHMAC(const AData: string): string;
var
  HMAC: TIdHMACSHA256;
  Key: string;
  DataBytes, KeyBytes: TIdBytes;
begin
  Key := GetDynamicKey + '_HMAC';
  KeyBytes := ToBytes(Key, IndyTextEncoding_UTF8);
  DataBytes := ToBytes(AData, IndyTextEncoding_UTF8);
  
  HMAC := TIdHMACSHA256.Create;
  try
    HMAC.Key := KeyBytes;
    Result := HMAC.HashBytesAsHex(DataBytes);
  finally
    HMAC.Free;
  end;
end;

// 数据完整性验证
class function TBasicProtection.VerifyDataIntegrity(const AData, AHMAC: string): Boolean;
begin
  Result := SameText(CalculateHMAC(AData), AHMAC);
end;

// 文件哈希计算 (使用Delphi内置THashSHA2)
class function TBasicProtection.CalculateFileHash(const AFileName: string): string;
var
  FileStream: TFileStream;
  Hash: THashSHA2;
  HashBytes: TBytes;
begin
  if not FileExists(AFileName) then
    raise Exception.Create('文件不存在: ' + AFileName);

  FileStream := TFileStream.Create(AFileName, fmOpenRead or fmShareDenyWrite);
  try
    Hash := THashSHA2.Create(SHA256);
    try
      Hash.Update(FileStream);
      HashBytes := Hash.HashAsBytes;
      Result := BytesToHex(HashBytes);
    finally
      Hash.Free;
    end;
  finally
    FileStream.Free;
  end;
end;

end.

// 程序启动时的基础检查
function PerformBasicSelfCheck: Boolean;
var
  ExeHash, StoredHash: string;
  ConfigData, ConfigHMAC: string;
begin
  Result := True;

  try
    // 1. 检查主程序文件哈希
    ExeHash := TBasicProtection.CalculateFileHash(Application.ExeName);
    StoredHash := LoadStoredHash; // 从配置或资源加载
    if (StoredHash <> '') and not SameText(ExeHash, StoredHash) then
    begin
      MessageBox(0, '程序文件可能被修改，请重新下载', '安全警告', MB_OK or MB_ICONWARNING);
      Result := False;
      Exit;
    end;

    // 2. 检查配置数据完整性
    ConfigData := LoadConfigData;
    ConfigHMAC := LoadConfigHMAC;
    if not TBasicProtection.VerifyDataIntegrity(ConfigData, ConfigHMAC) then
    begin
      MessageBox(0, '配置数据可能被篡改', '安全警告', MB_OK or MB_ICONWARNING);
      Result := False;
    end;

  except
    on E: Exception do
    begin
      // 记录错误日志
      LogError('自检失败: ' + E.Message);
      Result := False;
    end;
  end;
end;

// U89_LicenseProtection.pas - 纯代码实现
unit U89_LicenseProtection;

interface

uses
  Windows, SysUtils, Classes, Registry, WinSock;

type
  TLicenseInfo = record
    UserName: string;
    CompanyName: string;
    ProductCode: string;
    ExpiryDate: TDateTime;
    Features: TArray<string>;
    MachineFingerprint: string;
  end;

  TLicenseProtection = class
  private
    class function GetCPUSerialNumber: string;
    class function GetDiskSerialNumber: string;
    class function GetMACAddress: string;
    class function GenerateMachineFingerprint: string;
    class function CreateLicenseSignature(const ALicenseInfo: TLicenseInfo): string;
  public
    class function GenerateLicense(const ALicenseInfo: TLicenseInfo): string;
    class function ValidateLicense(const ALicenseData: string; out ALicenseInfo: TLicenseInfo): Boolean;
    class function IsFeatureEnabled(const ALicenseInfo: TLicenseInfo; const AFeature: string): Boolean;
  end;

implementation

// 获取CPU序列号
class function TLicenseProtection.GetCPUSerialNumber: string;
var
  Reg: TRegistry;
begin
  Result := '';
  Reg := TRegistry.Create(KEY_READ);
  try
    Reg.RootKey := HKEY_LOCAL_MACHINE;
    if Reg.OpenKey('HARDWARE\DESCRIPTION\System\CentralProcessor\0', False) then
    begin
      if Reg.ValueExists('ProcessorNameString') then
        Result := Reg.ReadString('ProcessorNameString');
      Reg.CloseKey;
    end;
  finally
    Reg.Free;
  end;

  // 如果无法获取，使用备用方法
  if Result = '' then
    Result := 'CPU_' + IntToStr(GetTickCount);
end;

// 获取磁盘序列号
class function TLicenseProtection.GetDiskSerialNumber: string;
var
  VolumeSerialNumber: DWORD;
  MaxComponentLength: DWORD;
  FileSystemFlags: DWORD;
  VolumeNameBuffer: array[0..MAX_PATH-1] of Char;
  FileSystemNameBuffer: array[0..MAX_PATH-1] of Char;
begin
  Result := '';

  if GetVolumeInformation(
    PChar('C:\'),
    VolumeNameBuffer,
    SizeOf(VolumeNameBuffer),
    @VolumeSerialNumber,
    MaxComponentLength,
    FileSystemFlags,
    FileSystemNameBuffer,
    SizeOf(FileSystemNameBuffer)) then
  begin
    Result := IntToHex(VolumeSerialNumber, 8);
  end
  else
    Result := 'DISK_' + IntToStr(GetTickCount);
end;

// 获取MAC地址
class function TLicenseProtection.GetMACAddress: string;
var
  WSAData: TWSAData;
  HostName: array[0..255] of AnsiChar;
  HostEnt: PHostEnt;
  Addr: TInAddr;
begin
  Result := '';

  if WSAStartup($0101, WSAData) = 0 then
  try
    if gethostname(HostName, SizeOf(HostName)) = 0 then
    begin
      HostEnt := gethostbyname(HostName);
      if HostEnt <> nil then
      begin
        Addr.S_addr := Longint(PLongint(HostEnt^.h_addr_list^)^);
        Result := inet_ntoa(Addr);
      end;
    end;
  finally
    WSACleanup;
  end;

  // 如果无法获取，使用备用标识
  if Result = '' then
    Result := 'MAC_' + IntToStr(GetTickCount);
end;

// 生成机器指纹
class function TLicenseProtection.GenerateMachineFingerprint: string;
var
  CPUInfo, DiskInfo, MACInfo: string;
  Combined: string;
begin
  CPUInfo := GetCPUSerialNumber;
  DiskInfo := GetDiskSerialNumber;
  MACInfo := GetMACAddress;

  Combined := CPUInfo + '|' + DiskInfo + '|' + MACInfo;
  Result := TBasicProtection.CalculateHMAC(Combined);
end;

// 许可证签名生成
class function TLicenseProtection.CreateLicenseSignature(const ALicenseInfo: TLicenseInfo): string;
var
  DataToSign: string;
  SignatureKey: string;
begin
  DataToSign := ALicenseInfo.UserName + '|' +
                ALicenseInfo.ProductCode + '|' +
                DateTimeToStr(ALicenseInfo.ExpiryDate) + '|' +
                ALicenseInfo.MachineFingerprint;

  SignatureKey := 'LicenseSignKey2024';
  Result := TBasicProtection.CalculateHMAC(DataToSign + SignatureKey);
end;

// 生成许可证
class function TLicenseProtection.GenerateLicense(const ALicenseInfo: TLicenseInfo): string;
var
  LicenseData: string;
  Signature: string;
  EncryptedLicense: string;
begin
  // 构建许可证数据
  LicenseData := Format('%s|%s|%s|%s|%s|%s',
    [ALicenseInfo.UserName,
     ALicenseInfo.CompanyName,
     ALicenseInfo.ProductCode,
     DateTimeToStr(ALicenseInfo.ExpiryDate),
     String.Join(',', ALicenseInfo.Features),
     ALicenseInfo.MachineFingerprint]);

  // 生成签名
  Signature := CreateLicenseSignature(ALicenseInfo);

  // 加密许可证数据
  EncryptedLicense := TBasicProtection.EncryptSensitiveData(LicenseData + '|' + Signature);

  Result := EncryptedLicense;
end;

// 验证许可证
class function TLicenseProtection.ValidateLicense(const ALicenseData: string; out ALicenseInfo: TLicenseInfo): Boolean;
var
  DecryptedData: string;
  Parts: TArray<string>;
  StoredSignature, CalculatedSignature: string;
  CurrentFingerprint: string;
begin
  Result := False;

  try
    // 解密许可证数据
    DecryptedData := TBasicProtection.DecryptSensitiveData(ALicenseData);

    // 分解许可证数据
    Parts := DecryptedData.Split(['|']);
    if Length(Parts) <> 7 then
      Exit;

    // 填充许可证信息
    ALicenseInfo.UserName := Parts[0];
    ALicenseInfo.CompanyName := Parts[1];
    ALicenseInfo.ProductCode := Parts[2];
    ALicenseInfo.ExpiryDate := StrToDateTime(Parts[3]);
    ALicenseInfo.Features := Parts[4].Split([',']);
    ALicenseInfo.MachineFingerprint := Parts[5];
    StoredSignature := Parts[6];

    // 验证签名
    CalculatedSignature := CreateLicenseSignature(ALicenseInfo);
    if not SameText(StoredSignature, CalculatedSignature) then
      Exit;

    // 验证机器指纹
    CurrentFingerprint := GenerateMachineFingerprint;
    if not SameText(ALicenseInfo.MachineFingerprint, CurrentFingerprint) then
      Exit;

    // 验证过期时间
    if ALicenseInfo.ExpiryDate < Now then
      Exit;

    Result := True;

  except
    Result := False;
  end;
end;

// 检查功能是否启用
class function TLicenseProtection.IsFeatureEnabled(const ALicenseInfo: TLicenseInfo; const AFeature: string): Boolean;
var
  Feature: string;
begin
  Result := False;
  for Feature in ALicenseInfo.Features do
  begin
    if SameText(Feature, AFeature) or SameText(Feature, 'ALL') then
    begin
      Result := True;
      Break;
    end;
  end;
end;

end.

procedure TTrialManager.LoadTrialConfig;

  Config: TConfigSQLite;
  EncryptedStartDate: string;
begin
  Config := TConfigSQLite.Create(FTrialConfigFile);
  try
    // 从加密配置中读取试用信息
    EncryptedStartDate := Config.GetString('Trial', 'StartDate', '');

        FUsageCount := 0;
        SaveTrialConfig;
      end;

  end;

procedure TTrialManager.SaveTrialConfig;
var
  Config: TConfigSQLite;
  EncryptedStartDate: string;
begin
  Config := TConfigSQLite.Create(FTrialConfigFile);
  try
    EncryptedStartDate := TBasicProtection.EncryptSensitiveData(DateTimeToStr(FTrialStartDate));
    Config.SetString('Trial', 'StartDate', EncryptedStartDate, '试用开始日期');
    Config.SetInteger('Trial', 'UsageCount', FUsageCount, '使用次数');
  finally
    Config.Free;
  end;
end;

function TTrialManager.IsTrialValid: Boolean;
var
  CurrentDate: TDateTime;
  DaysUsed: Integer;
begin
  CurrentDate := Now;
  DaysUsed := DaysBetween(CurrentDate, FTrialStartDate);

  Result := (DaysUsed <= FTrialDays) and (FUsageCount <= FMaxUsageCount);
end;

function TTrialManager.GetRemainingDays: Integer;
var
  DaysUsed: Integer;
begin
  DaysUsed := DaysBetween(Now, FTrialStartDate);
  Result := FTrialDays - DaysUsed;
  if Result < 0 then
    Result := 0;
end;

function TTrialManager.CanUseFeature(const AFeature: string): Boolean;
begin
  // 试用版功能限制
  if not IsTrialValid then
  begin
    Result := False;
    Exit;
  end;

  case AFeature of
    'export': Result := GetRemainingDays > 7; // 导出功能需要7天以上试用期
    'advanced': Result := False; // 高级功能试用版不可用
    'basic': Result := True; // 基础功能可用
    'save': Result := FUsageCount < (FMaxUsageCount div 2); // 保存功能限制使用次数
  else
    Result := True;
  end;
end;

procedure TTrialManager.RecordUsage;
begin
  Inc(FUsageCount);
  SaveTrialConfig;
end;

### **安全级别对比**

| 场景 | 安全级别 | 实现复杂度 | 性能影响 | 第三方依赖 |
|------|---------|-----------|---------|-----------|
| 个人软件 | ⭐⭐⭐ | 低 | 最小 (50-100ms) | 无 |
| 商业软件 | ⭐⭐⭐⭐ | 中 | 轻微 (100-200ms) | 无 |
| 企业软件 | ⭐⭐⭐⭐⭐ | 中-高 | 中等 (200-300ms) | 无 |

### **实施建议**

#### **选择原则**
1. **够用即可**: 根据实际威胁级别选择合适方案
2. **纯代码实现**: 避免第三方组件依赖问题
3. **渐进式**: 从基础保护开始，逐步增强
4. **可维护**: 代码简洁，易于理解和维护

#### **技术优势**
✅ **无依赖**: 仅使用Windows CryptoAPI和Delphi内置功能
✅ **稳定可靠**: 基于系统原生API，兼容性好
✅ **安全性高**: AES-256-CBC + HMAC-SHA256 标准加密
✅ **性能优化**: 针对Delphi优化的实现
✅ **易于维护**: 纯Pascal代码，便于调试和修改

#### **注意事项**
⚠️ **Windows专用**: 依赖Windows CryptoAPI，仅支持Windows平台
⚠️ **系统要求**: 需要Windows 7+支持AES-256算法
⚠️ **密钥管理**: 动态密钥生成，避免硬编码
⚠️ **错误处理**: 完善的异常处理和错误恢复机制
