## 🔧 技术实现详情

### 初始化逻辑
```pascal
procedure TfrmMain.InitializeShellTreeViews;
var
  SourceUsersPath, TargetUsersPath: string;
begin
  // 设置默认路径
  SourceUsersPath := 'C:\Users';
  TargetUsersPath := 'D:\Users';
  
  // 智能创建目标目录
  if not System.SysUtils.DirectoryExists(TargetUsersPath) then
  begin
    try
      System.SysUtils.ForceDirectories(TargetUsersPath);
      UpdateStatus('已创建目标目录: ' + TargetUsersPath);
    except
      on E: Exception do
      begin
        UpdateStatus('创建目标目录失败: ' + E.Message);
        TargetUsersPath := 'D:\'; // 智能回退
      end;
    end;
  end;
  
  // 同步更新所有相关控件
  FSourcePath := SourceUsersPath;
  FTargetPath := TargetUsersPath;
  edtSourceDir.Text := FSourcePath;
  edtTargetDir.Text := FTargetPath;
  
  // 导航到指定目录
  UpdateShellTreePath(stvSource, FSourcePath);
  UpdateShellTreePath(stvTarget, FTargetPath);
end;
```

### 实时同步机制
```pascal
procedure TfrmMain.stvSourceChange(Sender: TObject; Node: TTreeNode);
var
  SelectedPath: string;
begin
  try
    if Assigned(Node) and Assigned(stvSource.Selected) then
    begin
      SelectedPath := stvSource.Path;
      if SelectedPath <> '' then
      begin
        // 四重同步更新
        FSourcePath := SelectedPath;              // 1. 内部变量
        edtSourceDir.Text := SelectedPath;        // 2. 编辑框显示
        UpdateStatus('源目录已选择: ' + SelectedPath); // 3. 状态消息
        
        // 4. 状态栏显示
        if StatusBar1.Panels.Count > 1 then
          StatusBar1.Panels[1].Text := '源目录: ' + ExtractFileName(SelectedPath);
      end;
    end;
  except
    on E: Exception do
      UpdateStatus('源目录选择出错: ' + E.Message);
  end;
end;
```
